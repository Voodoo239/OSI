#!/bin/bash

# Инициализация полей
field=($(printf '0 %.0s' {1..20}))
enemy_field=($(printf '0 %.0s' {1..20}))
enemy_hits=0
my_hits=0
enemy_ships_total=9
enemy_ships_left=9
checked=($(printf '0 %.0s' {1..20}))

# Счетчики потопленных кораблей
enemy_1deck=1
enemy_2deck=1
enemy_3deck=2

# Функция для размещения кораблей
place_ships() {
    pos=$((RANDOM % 20))
    field[$pos]=1
    while true; do
        pos=$((RANDOM % 19))
        if [ ${field[$pos]} -eq 0 ] && [ ${field[$((pos+1))]} -eq 0 ]; then
            field[$pos]=1
            field[$((pos+1))]=1
            break
        fi
    done
    while true; do
        pos=$((RANDOM % 18))
        if [ ${field[$pos]} -eq 0 ] && [ ${field[$((pos+1))]} -eq 0 ] && [ ${field[$((pos+2))]} -eq 0 ]; then
            field[$pos]=1
            field[$((pos+1))]=1
            field[$((pos+2))]=1
            break
        fi
    done
    while true; do
        pos=$((RANDOM % 18))
        if [ ${field[$pos]} -eq 0 ] && [ ${field[$((pos+1))]} -eq 0 ] && [ ${field[$((pos+2))]} -eq 0 ]; then
            field[$pos]=1
            field[$((pos+1))]=1
            field[$((pos+2))]=1
            break
        fi
    done
}

# Функция для проверки, потоплен ли корабль
check_sunk() {
    local pos=$1
    local size=1
    while [ $((pos - size)) -ge 0 ] && [ ${enemy_field[$((pos - size))]} -eq 2 ]; do
        ((size++))
    done
    while [ $((pos + size)) -lt 20 ] && [ ${enemy_field[$((pos + size))]} -eq 2 ]; do
        ((size++))
    done
    if [ $size -gt 1 ]; then
        case $size in
            2) ((enemy_1deck--)); echo "1-deck sunk" ;;
            3|4) ((enemy_2deck--)); echo "2-deck sunk" ;;
            5|6) ((enemy_3deck--)); echo "3-deck sunk" ;;
        esac
        enemy_ships_left=$((enemy_ships_left - size + 1))
    fi
}

# Функция для выбора выстрела
shoot() {
    if [ $enemy_ships_left -ge 3 ] && [ $((enemy_3deck + enemy_2deck)) -gt 0 ]; then
        for i in {0..19}; do
            if [ ${enemy_field[$i]} -eq 2 ]; then
                if [ $i -gt 0 ] && [ ${checked[$((i-1))]} -eq 0 ] && [ ${enemy_field[$((i-1))]} -ne 3 ]; then
                    checked[$((i-1))]=1
                    echo $((i-1))
                    check_sunk $((i-1))
                    return
                fi
                if [ $i -lt 19 ] && [ ${checked[$((i+1))]} -eq 0 ] && [ ${enemy_field[$((i+1))]} -ne 3 ]; then
                    checked[$((i+1))]=1
                    echo $((i+1))
                    check_sunk $((i+1))
                    return
                fi
            fi
        done
    fi
    while true; do
        pos=$((RANDOM % 20))
        if [ ${checked[$pos]} -eq 0 ] && [ ${enemy_field[$pos]} -ne 3 ]; then
            checked[$pos]=1
            echo $pos
            check_sunk $pos
            return
        fi
        all_checked=1
        for i in {0..19}; do
            if [ ${checked[$i]} -eq 0 ] && [ ${enemy_field[$i]} -ne 3 ]; then
                all_checked=0
                break
            fi
        done
        if [ $all_checked -eq 1 ]; then
            echo "Error: All valid cells checked but game not finished!" >> game.log
            echo "QUIT" > pipe2
            exit 1
        fi
    done
}

# Размещаем корабли
place_ships

# Основной цикл игры
echo "Player 2: My PID is $$" | tee -a game.log
while true; do
    read enemy_shot < pipe1
    if [ "$enemy_shot" = "QUIT" ]; then
        echo "Player 2: Received QUIT, exiting" | tee -a game.log
        break
    fi
    echo "Player 1 shoots at $enemy_shot" | tee -a game.log
    if [ ${field[$enemy_shot]} -eq 1 ]; then
        echo "hit" > pipe2
        field[$enemy_shot]=2
        my_hits=$((my_hits + 1))
        echo "Player 2: Hit on my field at $enemy_shot" | tee -a game.log
    else
        echo "miss" > pipe2
        field[$enemy_shot]=3
        echo "Player 2: Miss on my field at $enemy_shot" | tee -a game.log
    fi

    if [ $my_hits -eq 9 ]; then
        echo "Player 1 wins!" | tee -a game.log
        echo "QUIT" > pipe2
        break
    fi

    shot=$(shoot)
    echo "Player 2 shoots at $shot" | tee -a game.log
    echo "$shot" > pipe1

    read result < pipe2
    if [ "$result" = "QUIT" ]; then
        echo "Player 2: Received QUIT, exiting" | tee -a game.log
        break
    fi
    if [ "$result" = "hit" ]; then
        enemy_hits=$((enemy_hits + 1))
        enemy_field[$shot]=2
        echo "Player 2: Hit! Enemy ships left: $enemy_ships_left" | tee -a game.log
    else
        enemy_field[$shot]=3
        echo "Player 2: Miss" | tee -a game.log
    fi

    if [ $enemy_hits -eq 9 ]; then
        echo "Player 2 wins!" | tee -a game.log
        echo "QUIT" > pipe1
        break
    fi
done
